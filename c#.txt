    class Urun
    {
        public string isim;
        public int fiyat;
        public Urun()
        {

        }
        public Urun(string isim,int fiyat)
        {
            this.isim = isim;
            this.fiyat = fiyat;
        }
        
    }
    class Dvd:Urun
    {
        public string tur;
        public double sure;
        public Dvd()
        {

        }
        public Dvd(string isim, int fiyat,string tur,double sure):base(isim,fiyat)
        {
            this.tur = tur;
            this.sure = sure;
            this.fiyat = fiyat;
            this.isim = isim;


        }
        public override string ToString()
        {
            return string.Format("{0} isimli dvd {1} turunde olup {2} dkdýr fiyatý ise :{3} YTL dir",isim,tur,sure,fiyat);
        }
        
    }
    /*
     *T urun olsun die yani liste urun tipnde 
     *deger alacak
     *oradaki new() => new list<> diye yaptýgýmýz için
     */
    class Urunler<T> where T:Urun,new() 
    {
        List<T> liste;
        public Urunler()
        {
            liste = new List<T>();
        }
        public void Ekle(T urun)
        {
            liste.Add(urun);
        }
        public void Cýkar(T urun)
        {
            liste.Remove(urun);
        }
        public void Cýkar(int index)
        {
            liste.RemoveAt(index);
        }
        public void Listeler()
        {
            foreach (var item in liste)
            {
                Console.WriteLine(item);
            }
        }
        
    }
